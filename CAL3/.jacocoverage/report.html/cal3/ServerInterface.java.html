    public ServerInterface() {
        initComponents();
        log = new LoggerThread();
        supermarket = new Supermarket(jTextFieldAttendingBuyerButcher, jTextFieldAttendingBuyerFishmonger, jTextFieldButcherShopQueue, jTextFieldFishShopQueue, jTextFieldBuyersShelves, jTextFieldNumberPeopleShelves, jTextFieldCashier1AttendingBuyer, jTextFieldCashier2AttendingBuyer, jTextFieldCheckAreaQueue, jTextFieldOusideQueue, monitor, log);
        while(counter <= 100)
        {
            //System.out.println("hi");
            if(!monitor.isStopThread())
            {
                //System.out.println("Entering " + counter);
                    try
                    {
                        Sleep( (800*Math.random() + 200));
                        buyer = new Buyer("Buyer " + counter, monitor, supermarket, 100, log);
                    }
                    catch(Exception e)
                    {
                    }
                
                counter++;
            }
            else
            {
                monitor.waitResume();
            }
        }
    }
